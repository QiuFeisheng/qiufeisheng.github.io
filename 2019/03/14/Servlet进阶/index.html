Servlet配置方式

    1. 全路径匹配

[Demo本]

  </servlet-mapping>

//题解

Servlet的配置方式1：Src—>New—>Servlet—>系统自动在web.xml自动配置Servlet的基本信息;

作用类似类似Android的Manifest自动注册和配置；

我们可以改url-pattern为/a 或者 /aa/bb都行，只要以/开头就行；

访问方式：localhost:8080/项目名称/a

    1. 路径匹配 , 前半段匹配

[Demo本]

<servlet>
<description></description>
<display-name>MyServlet</display-name>
<servlet-name>MyServlet</servlet-name>
<servlet-class>com.qiufeisheng.servlet.MyServlet</servlet-class>
  </servlet>

  <servlet-mapping>

批注：以/ 开始 ， 但是以 * 结束，例如：/a/*，/* 都符合条件

‘*’其实是一个通配符，匹配任意文字

访问方式：localhost:8080/项目名称/aa/xxxxxxxx/xxxx/xxx,都没问题，都可以访问

import java.io.IOException;
import javax.servlet.ServletContext;
import javax.servlet.ServletException;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;


public class MyServlet extends HttpServlet {
private static final long serialVersionUID = 1L;


public MyServlet() {
    super();
}


protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {

    //获取对象，调用象方：核心本质思想永远不会变
    ServletContext context=getServletContext();
    String number=context.getInitParameter("phonenumber");
    System.out.println("这是01Servlet获取的全局参数值=="+number);
}


protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {

    doGet(request, response);
    }
}

访问：http://localhost:8080/TestServlet/a/bb

import java.io.FileInputStream;
import java.io.IOException;
import java.io.InputStream;
import java.util.Properties;
import javax.servlet.ServletContext;
import javax.servlet.ServletException;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

public class MyServlet extends HttpServlet {

protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {

    //构造ServletContext来获取File的绝对路径（就是真实路径）
    ServletContext context=getServletContext();
    String path = context.getRealPath("File/config.properties");

    //加载目标文件，构造Properties对象
    Properties pro=new Properties();
    InputStream is=new FileInputStream(path);
    pro.load(is);

    String number=pro.getProperty("PhoneNumber");
    System.out.println("MyNumber is "+number);
}


protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {

    doGet(request, response);
}

}

String path = context.getRealPath(“file/config.properties”); //相对路径加载不了，只能获取绝对路径才能Load（真实路劲）

path的值是这个真实路径：D:\tomcat\apache-tomcat-7.0.52\apache-tomcat-7.0.52\wtpwebapps\Demo03\

访问：http://localhost:8080/TestServlet/a/bb就可以看到效果

（2）login.xml的Servlet: LoginServlet.java

import java.io.IOException;
import java.io.PrintWriter;
import javax.servlet.ServletContext;
import javax.servlet.ServletException;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;


public class LoginServlet extends HttpServlet {

protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {

    //1.获取客户端提交的数据
    String username=request.getParameter("username");
    String password=request.getParameter("password");


    //get请求过来的数据，在url地址栏上就已经经过编码了，所以我们取到的就是乱码，
    //tomcat收到了这批数据，getParameter 默认使用ISO-8859-1去解码
    //先让文字回到ISO-8859-1对应的字节数组 ， 然后再按utf-8组拼字符串
    username = new String(username.getBytes("ISO-8859-1") , "UTF-8");
    System.out.println("userName="+username+"==password="+password);


    //输出响应
    System.out.println("username=="+username+"  "+"password=="+password);
    PrintWriter pw=response.getWriter();

    //构造ServletContext对象来存取登录次数，从而让登录成功页面的Servlet共享数据
    Object obj=getServletContext().getAttribute("count");
    int totalNumber=0;
    if(obj!=null) {
        totalNumber=(int)obj;
    }
    getServletContext().setAttribute("count", totalNumber+1);

    //html跳转
    if("admin".equals(username)&&"123".equals(password)) {

        //设置状态码302可以重定位网页资源
        response.setStatus(302);
        response.setHeader("Location","login_success.html");
    }else {

        //登录失败就直接在Servlet的页面输出响应，上面是跳转到别的html页面
        pw.write("login failed");
    }
}

protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {

    doGet(request, response);
}

}

(3)超链接到Servlet来完成下载操作

import java.io.FileInputStream;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;
import javax.servlet.ServletException;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;


public class DownloadServlet extends HttpServlet {


protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {

    //从html中拿到filename
    String filename=request.getParameter("filename");

    //file在Web中的资源路劲
    String path=getServletContext().getRealPath("download/"+filename);

    //设置动作类型，点击后即可下载
    response.setHeader("Content-Disposition", "attachment;filename="+filename);



    InputStream is=new FileInputStream(path);
    OutputStream os=response.getOutputStream();

    int len=0;
    byte[] buffer=new byte[1024];
    while((len=is.read(buffer))!=-1) {
        os.write(buffer,0,len);
    }

    os.close();
    is.close();
}


protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {

    doGet(request, response);
        }

    }
